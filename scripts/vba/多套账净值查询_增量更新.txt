Option Explicit

' ==========================================================
' 功能：基于“多套账净值查询.xlsx”中的最新日期，检查源目录是否有更新文件，
'       如有则把新文件中的有效数据增量导入到“多套账净值查询.xlsx”。
'
' 约定/假设（可按需修改）：
' (1) 目标工作簿：d:\bocom\nav_open_day_vba\多套账净值查询.xlsx
' (2) 目标工作表：优先使用“合并数据”，若不存在则使用第1个工作表
' (3) 日期列：B列（yyyyMMdd），用于判断最新日期与定位数据起始行
' (4) 源文件目录：d:\bocom\nav_open_day_vba\181-固收上层
' (5) 源文件名中包含yyyyMMdd（8位数字），用于判断是否为新文件
' (6) 源文件数据：每个工作表均可能有数据；脚本会尝试自动识别表头行（扫描前30行）
' (7) 导入方式：按目标表头字段名做列映射，仅导入表头行下方的数据行；追加写入到目标表末尾
'
' 注意：该宏不会修改/删除源目录文件，仅打开读取后关闭。
' ==========================================================

Private Const TARGET_WB_PATH As String = "d:\bocom\nav_open_day_vba\多套账净值查询.xlsx"
Private Const SOURCE_DIR As String = "d:\bocom\nav_open_day_vba\181-固收上层"
Private Const TARGET_SHEET_NAME As String = "合并数据"
Private Const DATE_COL As Long = 2 ' B列


Public Sub 增量导入_多套账净值查询()
    Dim wbTarget As Workbook
    Dim wsTarget As Worksheet
    Dim dataStartRow As Long
    Dim latestYmd As Long
    Dim newFiles As Collection
    Dim i As Long

    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    Application.EnableEvents = False

    On Error GoTo CleanFail

    Set wbTarget = Workbooks.Open(Filename:=TARGET_WB_PATH, ReadOnly:=False)
    Set wsTarget = GetTargetSheet(wbTarget)

    dataStartRow = FindFirstYyyymmddRow(wsTarget, DATE_COL)
    If dataStartRow = 0 Then
        Err.Raise vbObjectError + 100, , "在目标文件B列未找到任何yyyyMMdd格式日期，无法定位数据起始行。"
    End If

    latestYmd = FindMaxYyyymmddInColumn(wsTarget, DATE_COL, dataStartRow)
    If latestYmd = 0 Then
        Err.Raise vbObjectError + 101, , "在目标文件B列未解析到有效yyyyMMdd日期，无法计算最新日期。"
    End If

    Set newFiles = ListNewFilesByDate(SOURCE_DIR, latestYmd)
    If newFiles.Count = 0 Then
        MsgBox "未发现比当前最新日期更大的新文件。最新日期=" & CStr(latestYmd), vbInformation
        GoTo CleanOK
    End If

    ImportFilesAppend wbTarget:=wbTarget, wsTarget:=wsTarget, dataStartRow:=dataStartRow, files:=newFiles

    wbTarget.Save
    MsgBox "增量导入完成。新增文件数=" & CStr(newFiles.Count) & "，最新日期=" & CStr(latestYmd), vbInformation

CleanOK:
    On Error Resume Next
    wbTarget.Close SaveChanges:=True
    Application.EnableEvents = True
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
    Exit Sub

CleanFail:
    Dim msg As String
    msg = "增量导入失败：" & Err.Description
    MsgBox msg, vbCritical
    On Error Resume Next
    If Not wbTarget Is Nothing Then wbTarget.Close SaveChanges:=False
    Application.EnableEvents = True
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
End Sub


Private Function GetTargetSheet(ByVal wb As Workbook) As Worksheet
    On Error Resume Next
    Set GetTargetSheet = wb.Worksheets(TARGET_SHEET_NAME)
    On Error GoTo 0
    If GetTargetSheet Is Nothing Then
        Set GetTargetSheet = wb.Worksheets(1)
    End If
End Function


' 在指定列中找到第一个满足yyyyMMdd的单元格所在行（用于定位数据起始行）
Private Function FindFirstYyyymmddRow(ByVal ws As Worksheet, ByVal colIndex As Long) As Long
    Dim lastRow As Long
    Dim r As Long
    Dim ymd As Long

    lastRow = ws.Cells(ws.Rows.Count, colIndex).End(xlUp).Row
    For r = 1 To lastRow
        ymd = ParseYyyymmddValue(ws.Cells(r, colIndex).Value2)
        If ymd <> 0 Then
            FindFirstYyyymmddRow = r
            Exit Function
        End If
    Next r
    FindFirstYyyymmddRow = 0
End Function


' 从指定列中提取最大yyyyMMdd（只从dataStartRow开始扫描）
Private Function FindMaxYyyymmddInColumn(ByVal ws As Worksheet, ByVal colIndex As Long, ByVal dataStartRow As Long) As Long
    Dim lastRow As Long
    Dim r As Long
    Dim ymd As Long
    Dim maxYmd As Long

    lastRow = ws.Cells(ws.Rows.Count, colIndex).End(xlUp).Row
    maxYmd = 0
    For r = dataStartRow To lastRow
        ymd = ParseYyyymmddValue(ws.Cells(r, colIndex).Value2)
        If ymd > maxYmd Then maxYmd = ymd
    Next r
    FindMaxYyyymmddInColumn = maxYmd
End Function


' 列出目录下所有日期大于latestYmd的文件（按日期升序返回）
Private Function ListNewFilesByDate(ByVal folderPath As String, ByVal latestYmd As Long) As Collection
    Dim fso As Object
    Dim folder As Object
    Dim file As Object
    Dim items As Collection
    Dim temp As Collection
    Dim i As Long, j As Long
    Dim cur As Variant, nxt As Variant

    Set fso = CreateObject("Scripting.FileSystemObject")
    Set folder = fso.GetFolder(folderPath)
    Set items = New Collection

    For Each file In folder.Files
        Dim ymd As Long
        ymd = ExtractYyyymmddFromText(CStr(file.Name))
        If ymd > latestYmd Then
            items.Add Array(ymd, CStr(file.Path))
        End If
    Next file

    Set temp = New Collection
    For i = 1 To items.Count
        temp.Add items(i)
    Next i

    For i = 1 To temp.Count - 1
        For j = i + 1 To temp.Count
            cur = temp(i)
            nxt = temp(j)
            If CLng(cur(0)) > CLng(nxt(0)) Then
                temp.Remove j
                temp.Add cur, , j
                temp.Remove i
                temp.Add nxt, , i
            End If
        Next j
    Next i

    Set ListNewFilesByDate = New Collection
    For i = 1 To temp.Count
        cur = temp(i)
        ListNewFilesByDate.Add CStr(cur(1))
    Next i
End Function


' 把files中的源文件内容追加导入到目标表末尾（按表头字段名做列映射）
Private Sub ImportFilesAppend(ByVal wbTarget As Workbook, ByVal wsTarget As Worksheet, ByVal dataStartRow As Long, ByVal files As Collection)
    Dim headerRowTarget As Long
    Dim dictTargetHeader As Object
    Dim lastColTarget As Long
    Dim lastRowTarget As Long
    Dim nextRow As Long
    Dim i As Long

    headerRowTarget = dataStartRow - 1
    If headerRowTarget < 1 Then
        Err.Raise vbObjectError + 200, , "目标表的数据起始行异常，无法定位表头行。"
    End If

    lastColTarget = wsTarget.Cells(headerRowTarget, wsTarget.Columns.Count).End(xlToLeft).Column
    Set dictTargetHeader = BuildHeaderDict(wsTarget, headerRowTarget, lastColTarget)

    lastRowTarget = wsTarget.Cells(wsTarget.Rows.Count, 1).End(xlUp).Row
    nextRow = lastRowTarget + 1

    For i = 1 To files.Count
        ImportOneFileAppend wsTarget:=wsTarget, dictTargetHeader:=dictTargetHeader, headerRowTarget:=headerRowTarget, nextRow:=nextRow, filePath:=CStr(files(i))
    Next i
End Sub


Private Sub ImportOneFileAppend(ByVal wsTarget As Worksheet, ByVal dictTargetHeader As Object, ByVal headerRowTarget As Long, ByRef nextRow As Long, ByVal filePath As String)
    Dim wbSrc As Workbook
    Dim wsSrc As Worksheet

    On Error GoTo ImportFail
    Set wbSrc = Workbooks.Open(Filename:=filePath, ReadOnly:=True)

    For Each wsSrc In wbSrc.Worksheets
        ImportOneSheetAppend wsTarget:=wsTarget, dictTargetHeader:=dictTargetHeader, headerRowTarget:=headerRowTarget, wsSrc:=wsSrc, nextRow:=nextRow, sourceFilePath:=filePath
    Next wsSrc

ImportDone:
    On Error Resume Next
    wbSrc.Close SaveChanges:=False
    Exit Sub

ImportFail:
    Dim msg As String
    msg = "导入源文件失败：" & vbCrLf & filePath & vbCrLf & Err.Description
    Debug.Print msg
    On Error Resume Next
    If Not wbSrc Is Nothing Then wbSrc.Close SaveChanges:=False
End Sub


Private Sub ImportOneSheetAppend(ByVal wsTarget As Worksheet, ByVal dictTargetHeader As Object, ByVal headerRowTarget As Long, ByVal wsSrc As Worksheet, ByRef nextRow As Long, ByVal sourceFilePath As String)
    Dim headerRowSrc As Long
    Dim lastRowSrc As Long
    Dim lastColSrc As Long
    Dim dictSrcHeader As Object
    Dim r As Long

    headerRowSrc = DetectHeaderRow(wsSrc, 30)
    If headerRowSrc <= 0 Then Exit Sub

    lastRowSrc = wsSrc.Cells(wsSrc.Rows.Count, 1).End(xlUp).Row
    lastColSrc = wsSrc.Cells(headerRowSrc, wsSrc.Columns.Count).End(xlToLeft).Column
    If lastRowSrc <= headerRowSrc Then Exit Sub

    Set dictSrcHeader = BuildHeaderDict(wsSrc, headerRowSrc, lastColSrc)

    For r = headerRowSrc + 1 To lastRowSrc
        If RowIsAllBlank(wsSrc, r, lastColSrc) Then
            GoTo NextR
        End If

        Dim key As Variant
        For Each key In dictTargetHeader.Keys
            Dim tgtCol As Long
            Dim srcCol As Long
            tgtCol = CLng(dictTargetHeader(key))
            If dictSrcHeader.Exists(CStr(key)) Then
                srcCol = CLng(dictSrcHeader(CStr(key)))
                wsTarget.Cells(nextRow, tgtCol).Value2 = wsSrc.Cells(r, srcCol).Value2
            End If
        Next key

        nextRow = nextRow + 1
NextR:
    Next r
End Sub


' 自动识别表头行：扫描前maxScanRows行，取“非空单元格数”最多的行
Private Function DetectHeaderRow(ByVal ws As Worksheet, ByVal maxScanRows As Long) As Long
    Dim r As Long
    Dim bestRow As Long
    Dim bestScore As Long
    Dim lastCol As Long
    Dim nonBlank As Long

    bestRow = 0
    bestScore = -1

    For r = 1 To maxScanRows
        lastCol = ws.Cells(r, ws.Columns.Count).End(xlToLeft).Column
        If lastCol < 1 Then GoTo NextR

        nonBlank = Application.WorksheetFunction.CountA(ws.Range(ws.Cells(r, 1), ws.Cells(r, lastCol)))
        If nonBlank >= 2 Then
            If nonBlank > bestScore Then
                bestScore = nonBlank
                bestRow = r
            End If
        End If
NextR:
    Next r

    DetectHeaderRow = bestRow
End Function


Private Function BuildHeaderDict(ByVal ws As Worksheet, ByVal headerRow As Long, ByVal lastCol As Long) As Object
    Dim dict As Object
    Dim c As Long
    Dim name As String

    Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = 1 ' vbTextCompare

    For c = 1 To lastCol
        name = NormalizeHeader(ws.Cells(headerRow, c).Value2)
        If Len(name) > 0 Then
            If Not dict.Exists(name) Then
                dict.Add name, c
            End If
        End If
    Next c

    Set BuildHeaderDict = dict
End Function


Private Function NormalizeHeader(ByVal v As Variant) As String
    Dim s As String
    s = Trim(CStr(v))
    s = Replace(s, vbCr, " ")
    s = Replace(s, vbLf, " ")
    s = Replace(s, ChrW(&H3000), " ")
    Do While InStr(s, "  ") > 0
        s = Replace(s, "  ", " ")
    Loop
    NormalizeHeader = Trim(s)
End Function


Private Function RowIsAllBlank(ByVal ws As Worksheet, ByVal rowIndex As Long, ByVal lastCol As Long) As Boolean
    RowIsAllBlank = (Application.WorksheetFunction.CountA(ws.Range(ws.Cells(rowIndex, 1), ws.Cells(rowIndex, lastCol))) = 0)
End Function


' 解析单元格值为yyyyMMdd（支持数值/文本），失败返回0
Private Function ParseYyyymmddValue(ByVal v As Variant) As Long
    Dim s As String
    s = Trim(CStr(v))
    If Len(s) = 0 Then
        ParseYyyymmddValue = 0
        Exit Function
    End If
    If IsNumeric(s) Then
        If Len(s) = 8 Then
            ParseYyyymmddValue = CLng(s)
        Else
            ParseYyyymmddValue = 0
        End If
    Else
        ParseYyyymmddValue = ExtractYyyymmddFromText(s)
    End If
End Function


' 从文本中提取首个合法yyyyMMdd（8位数字且可转为日期），失败返回0
Private Function ExtractYyyymmddFromText(ByVal text As String) As Long
    Dim i As Long
    Dim candidate As String
    Dim y As Integer, m As Integer, d As Integer

    For i = 1 To Len(text) - 7
        candidate = Mid$(text, i, 8)
        If IsAllDigits(candidate) Then
            y = CInt(Left$(candidate, 4))
            m = CInt(Mid$(candidate, 5, 2))
            d = CInt(Right$(candidate, 2))
            If IsValidDateParts(y, m, d) Then
                ExtractYyyymmddFromText = CLng(candidate)
                Exit Function
            End If
        End If
    Next i

    ExtractYyyymmddFromText = 0
End Function


Private Function IsAllDigits(ByVal s As String) As Boolean
    Dim i As Long
    If Len(s) = 0 Then
        IsAllDigits = False
        Exit Function
    End If
    For i = 1 To Len(s)
        Dim ch As Integer
        ch = AscW(Mid$(s, i, 1))
        If ch < 48 Or ch > 57 Then
            IsAllDigits = False
            Exit Function
        End If
    Next i
    IsAllDigits = True
End Function


Private Function IsValidDateParts(ByVal y As Integer, ByVal m As Integer, ByVal d As Integer) As Boolean
    On Error GoTo Fail
    Dim dt As Date
    dt = DateSerial(y, m, d)
    IsValidDateParts = (Year(dt) = y And Month(dt) = m And Day(dt) = d)
    Exit Function
Fail:
    IsValidDateParts = False
End Function
